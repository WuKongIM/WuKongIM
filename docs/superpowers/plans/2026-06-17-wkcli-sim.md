```go
func parseRate(raw string) (Rate, error) {
	trimmed := strings.TrimSpace(raw)
	if strings.HasSuffix(trimmed, "/s") {
		trimmed = strings.TrimSuffix(trimmed, "/s")
	}
	value, err := strconv.ParseFloat(strings.TrimSpace(trimmed), 64)
	if err != nil || value <= 0 {
		return Rate{}, fmt.Errorf("invalid rate %q", raw)
	}
	return Rate{PerSecond: value}, nil
}

func parseByteSize(raw string) (int, error) {
	trimmed := strings.TrimSpace(raw)
	if trimmed == "" {
		return 0, fmt.Errorf("invalid size %q", raw)
	}
	upper := strings.ToUpper(trimmed)
	units := []struct {
		suffix string
		mult   float64
	}{
		{"KIB", 1024}, {"MIB", 1024 * 1024}, {"GIB", 1024 * 1024 * 1024},
		{"KB", 1000}, {"MB", 1000 * 1000}, {"GB", 1000 * 1000 * 1000},
		{"B", 1},
	}
	mult := float64(1)
	number := upper
	for _, unit := range units {
		if strings.HasSuffix(upper, unit.suffix) {
			mult = unit.mult
			number = strings.TrimSpace(strings.TrimSuffix(upper, unit.suffix))
			break
		}
	}
	value, err := strconv.ParseFloat(number, 64)
	if err != nil || value <= 0 {
		return 0, fmt.Errorf("invalid size %q", raw)
	}
	return int(value * mult), nil
}

func splitValues(values []string) []string {
	var out []string
	for _, value := range values {
		for _, part := range strings.Split(value, ",") {
			part = strings.TrimSpace(part)
			if part != "" {
				out = append(out, part)
			}
		}
	}
	return out
}

func dedupeValues(values []string) []string {
	seen := map[string]struct{}{}
	out := make([]string, 0, len(values))
	for _, value := range values {
		value = strings.TrimSpace(value)
		if value == "" {
			continue
		}
		if _, ok := seen[value]; ok {
			continue
		}
		seen[value] = struct{}{}
		out = append(out, value)
	}
	return out
}

func newRunID(now time.Time) (string, error) {
	var b [4]byte
	if _, err := rand.Read(b[:]); err != nil {
		return "", fmt.Errorf("generate run id: %w", err)
	}
	return "wkcli-sim-" + now.Format("20060102-150405") + "-" + hex.EncodeToString(b[:]), nil
}
```

```go
func NewCommand(deps command.Deps) *cobra.Command {

		Use:          "sim",
		Short:        "Run a long-lived internalv2 traffic simulator",
```

```go
package sim

import (
	"fmt"
	"sync/atomic"

	wkclient "github.com/WuKongIM/WuKongIM/pkg/client"
	"github.com/WuKongIM/WuKongIM/pkg/protocol/frame"
)

// Plan is the deterministic simulator topology.
type Plan struct {
	// Users are the online identities connected to the gateway.
	Users []User
	// Groups are the generated group channels used for traffic.
	Groups []Group
}

// User is one generated online identity.
type User struct {
	// UID is the generated user id.
	UID string
	// DeviceID is the generated device id.
	DeviceID string
}

// Group is one generated group channel and its deterministic sender rotation.
type Group struct {
	// ChannelID is the generated group channel id.
	ChannelID string
	// Subscribers are the generated member UIDs.
	Subscribers []string
	next        atomic.Uint64
}

func buildPlan(cfg Config) Plan {
	users := make([]User, 0, cfg.Users)
	for i := 0; i < cfg.Users; i++ {
		users = append(users, User{
```

```go
func (c *targetClient) preflight(ctx context.Context) (resolvedTarget, error) {
	out := resolvedTarget{Servers: append([]string(nil), c.servers...)}
	for _, server := range c.servers {
		if err := c.get(ctx, server, "/healthz", nil); err != nil {
			return resolvedTarget{}, fmt.Errorf("target %s /healthz failed: %w", server, err)
		}
		if err := c.get(ctx, server, "/readyz", nil); err != nil {
			return resolvedTarget{}, fmt.Errorf("target %s /readyz failed: %w", server, err)
		}
		var caps capabilitiesResponse
		if err := c.get(ctx, server, "/bench/v1/capabilities", &caps); err != nil {
			return resolvedTarget{}, fmt.Errorf("target %s bench capabilities unavailable; ensure WK_BENCH_API_ENABLE=true: %w", server, err)
		}
		if err := validateCapabilities(caps); err != nil {
			return resolvedTarget{}, fmt.Errorf("target %s bench capabilities missing required support: %w", server, err)
		}
		if caps.Limits.MaxBatchSize > 0 && (out.MaxBatchSize == 0 || caps.Limits.MaxBatchSize < out.MaxBatchSize) {
			out.MaxBatchSize = caps.Limits.MaxBatchSize
		}
		var capacity capacityTargetResponse
		if err := c.get(ctx, server, "/bench/v1/capacity-target", &capacity); err != nil {
			return resolvedTarget{}, fmt.Errorf("target %s capacity target failed: %w", server, err)
		}
		if strings.TrimSpace(capacity.Gateway.TCPAddr) != "" {
			out.GatewayTCPAddrs = append(out.GatewayTCPAddrs, strings.TrimSpace(capacity.Gateway.TCPAddr))
		}
	}
	out.GatewayTCPAddrs = dedupeValues(out.GatewayTCPAddrs)
	if len(out.GatewayTCPAddrs) == 0 {
		return resolvedTarget{}, fmt.Errorf("capacity target returned empty gateway.tcp_addr; set WK_EXTERNAL_TCPADDR or pass --gateway")
	}
	return out, nil
}

func validateCapabilities(caps capabilitiesResponse) error {
	var missing []string
	if !caps.Enabled {
		missing = append(missing, "enabled")
	}
	if caps.Version != "bench/v1" {
		missing = append(missing, "version bench/v1")
	}
	if !caps.Supports.ChannelsBatch {
		missing = append(missing, "channels_batch")
	}
	if !caps.Supports.ChannelSubscribersBatch {
		missing = append(missing, "channel_subscribers_batch")
	}
	if !caps.Supports.Snapshot {
		missing = append(missing, "snapshot")
	}
	if !contains(caps.Supports.ChannelTypes, "group") {
		missing = append(missing, "channel_types group")
	}
	if len(missing) > 0 {
		return fmt.Errorf(strings.Join(missing, ", "))
	}
	return nil
}

func (c *targetClient) setup(ctx context.Context, cfg Config, target resolvedTarget, plan Plan) error {
	batchSize := target.MaxBatchSize
	if batchSize <= 0 {
		batchSize = len(plan.Groups)
	}
	for start := 0; start < len(plan.Groups); start += batchSize {
		end := min(start+batchSize, len(plan.Groups))
		req := channelsRequest{RunID: cfg.RunID, BatchID: fmt.Sprintf("channels-%06d", start), Upsert: true}
		for _, group := range plan.Groups[start:end] {
			req.Channels = append(req.Channels, channelItem{ChannelID: group.ChannelID, ChannelType: 2, AllowStranger: true})
		}
		if err := c.postAny(ctx, "/bench/v1/channels", req, &mutationResponse{}); err != nil {
			return err
		}
	}
	for start := 0; start < len(plan.Groups); start += batchSize {
		end := min(start+batchSize, len(plan.Groups))
		req := subscribersRequest{RunID: cfg.RunID, BatchID: fmt.Sprintf("subscribers-%06d", start)}
		for _, group := range plan.Groups[start:end] {
			req.Items = append(req.Items, subscriberItem{ChannelID: group.ChannelID, ChannelType: 2, Subscribers: group.Subscribers})
		}
		if err := c.postAny(ctx, "/bench/v1/channels/subscribers", req, &subscribersResponse{}); err != nil {
			return err
		}
	}
	return nil
}

func (c *targetClient) postAny(ctx context.Context, path string, in any, out any) error {
	var errs []string
	for _, server := range c.servers {
		if err := c.do(ctx, http.MethodPost, server, path, in, out); err != nil {
			errs = append(errs, err.Error())
			continue
		}
		return nil
	}
	return fmt.Errorf("target post %s failed: %s", path, strings.Join(errs, "; "))
}

func (c *targetClient) do(ctx context.Context, method, server, path string, in any, out any) error {
	var body io.Reader
	if in != nil {
		data, err := json.Marshal(in)
		if err != nil {
			return err
		}
		body = bytes.NewReader(data)
	}
	req, err := http.NewRequestWithContext(ctx, method, strings.TrimRight(server, "/")+path, body)
	if err != nil {
		return err
	}
	if in != nil {
		req.Header.Set("Content-Type", "application/json")
	}
	if c.token != "" {
		req.Header.Set("Authorization", "Bearer "+c.token)
	}
	resp, err := c.http.Do(req)
	if err != nil {
		return err
	}
	defer resp.Body.Close()
	if resp.StatusCode < 200 || resp.StatusCode >= 300 {
		data, _ := io.ReadAll(io.LimitReader(resp.Body, 512))
		return fmt.Errorf("%s %s returned status %d: %s", method, path, resp.StatusCode, strings.TrimSpace(string(data)))
	}
	if out != nil {
		return json.NewDecoder(resp.Body).Decode(out)
	}
	_, _ = io.Copy(io.Discard, resp.Body)
	return nil
}

func contains(items []string, want string) bool {

		if item == want {
			return true
		}
```

```go
import (
	"context"
```

```go
	"errors"
	"net"
	"net/http"

type statusServer struct {
	addr   string
	status *statusModel
	server *http.Server
	mux    *http.ServeMux
}

func newStatusServer(addr string, status *statusModel) *statusServer {
	s := &statusServer{addr: addr, status: status, mux: http.NewServeMux()}
	s.mux.HandleFunc("/healthz", s.healthz)
	s.mux.HandleFunc("/status", s.statusz)
	s.server = &http.Server{Addr: addr, Handler: s.mux}
	return s
}

func (s *statusServer) start(ctx context.Context) error {
	ln, err := net.Listen("tcp", s.addr)
	if err != nil {
		return err
	}
	errCh := make(chan error, 1)
	go func() {
		err := s.server.Serve(ln)
		if errors.Is(err, http.ErrServerClosed) {
			err = nil
		}
		errCh <- err
	}()
	select {
	case <-ctx.Done():
		shutdownCtx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
		defer cancel()
		if err := s.server.Shutdown(shutdownCtx); err != nil {
			return err
		}
		return <-errCh
```

```go
package sim

import (
	"context"
	"fmt"
	"sync"
	"sync/atomic"
	"time"

	wkclient "github.com/WuKongIM/WuKongIM/pkg/client"
	"github.com/WuKongIM/WuKongIM/pkg/protocol/frame"
)

type targetAPI interface {
	preflight(context.Context) (resolvedTarget, error)
	setup(context.Context, Config, resolvedTarget, Plan) error
}

type simPool interface {
	Connect(context.Context, []wkclient.Identity) error
	SendBatch(context.Context, []wkclient.RoutedMessage) ([]wkclient.SendResult, error)
	Close() error
}

type poolFactory func(Config, []string, wkclient.Observer) (simPool, error)
type sleepFunc func(context.Context, time.Duration) error

const runtimeRetryBackoff = 2 * time.Second

// Runtime supervises setup, connections, traffic, retry, and shutdown.
type Runtime struct {
	Config  Config
	Status  *statusModel
	Target  targetAPI
	NewPool poolFactory
	Sleep   sleepFunc
}

func (r *Runtime) Run(ctx context.Context) error {
	cfg := r.Config
	status := r.Status
	if status == nil {
		status = newStatus(cfg.RunID)
	}
	if r.Sleep == nil {
		r.Sleep = sleepContext
	}
	if r.NewPool == nil {
		r.NewPool = newClientPool
	}
	target := r.Target
	if target == nil {
		target = newTargetClient(cfg.Servers, cfg.BenchToken)
	}
	if cfg.MaxRuntime > 0 {
		var cancel context.CancelFunc
		ctx, cancel = context.WithTimeout(ctx, cfg.MaxRuntime)
		defer cancel()
	}
	status.setState(statePreflighting)
	resolved, err := target.preflight(ctx)
	if err != nil {
		status.addSendErrors(0, err.Error())
		status.setState(stateStopped)
		return err
	}
	if len(cfg.Gateways) > 0 {
		resolved.GatewayTCPAddrs = cfg.Gateways
	}
	status.setTarget(resolved.Servers, resolved.GatewayTCPAddrs)
	plan := buildPlan(cfg)
	status.setTopology(len(plan.Users), len(plan.Groups), cfg.GroupMembers)
	status.setState(stateSettingUp)
	if err := target.setup(ctx, cfg, resolved, plan); err != nil {
		status.addSendErrors(0, err.Error())
		status.setState(stateStopped)
		return err
	}
	var seq atomic.Uint64
	for {
		status.setState(stateConnecting)
		observer := &clientObserver{status: status}
		pool, err := r.NewPool(cfg, resolved.GatewayTCPAddrs, observer)
		if err != nil {
			status.setState(stateStopped)
			return err
		}
		if err := pool.Connect(ctx, identitiesFromPlan(plan)); err != nil {
			_ = pool.Close()
			status.setState(stateStopped)
			return err
		}
		status.setState(stateRunning)
		err = r.runTraffic(ctx, pool, plan, &seq)
		_ = pool.Close()
		if ctx.Err() != nil {
			status.setState(stateStopped)
			return nil
		}
		status.addReconnect(err.Error())
		status.setState(stateRetrying)
		if sleepErr := r.Sleep(ctx, runtimeRetryBackoff); sleepErr != nil {
			status.setState(stateStopped)
			return nil
		}
	}
}

func (r *Runtime) runTraffic(ctx context.Context, pool simPool, plan Plan, seq *atomic.Uint64) error {
	cfg := r.Config
	interval := time.Duration(float64(time.Second) / cfg.RatePerGroup.PerSecond)
	if interval <= 0 {
		interval = time.Nanosecond
	}
	work := make(chan *Group)
	var wg sync.WaitGroup
	var firstErr atomic.Value
	for i := 0; i < cfg.Concurrency; i++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			for group := range work {
				msg := group.nextMessage(cfg, seq.Add(1))
				results, err := pool.SendBatch(ctx, []wkclient.RoutedMessage{msg})
				if err != nil {
					r.Status.addSendErrors(1, err.Error())
					firstErr.CompareAndSwap(nil, err)
					continue
				}
				if len(results) == 0 || results[0].ReasonCode != frame.ReasonSuccess {
					reason := "missing sendack"
					if len(results) > 0 {
						reason = results[0].ReasonCode.String()
					}
					r.Status.addSendErrors(1, reason)
					continue
				}
				r.Status.addMessagesSent(1)
			}
		}()
	}
	ticker := time.NewTicker(interval)
	defer ticker.Stop()
	defer func() {
		close(work)
		wg.Wait()
	}()
	for {
		select {
		case <-ctx.Done():
			return ctx.Err()
		case <-ticker.C:
			for i := range plan.Groups {
				select {
				case work <- &plan.Groups[i]:
				default:
					r.Status.addSendErrors(1, "sim send work queue full")
				}
			}
			if value := firstErr.Load(); value != nil {
				return value.(error)
			}
		}
	}
}

func newClientPool(cfg Config, gateways []string, observer wkclient.Observer) (simPool, error) {
	return wkclient.NewPool(wkclient.PoolConfig{
		Addrs: gateways,
		Client: wkclient.Config{
			OperationTimeout:      cfg.OperationTimeout,
			AckTimeout:            cfg.AckTimeout,
			AutoRecvAck:           true,
			MaxInflight:           cfg.Concurrency,
			SendQueueCapacity:     cfg.Concurrency * 4,
			InboundFrameBufferSize: cfg.Concurrency * 4,
			Observer:              observer,
		},
		ConnectRatePerSecond: cfg.ConnectRate,
	})
}

type clientObserver struct {
	status *statusModel
}

func (o *clientObserver) OnConnect(wkclient.ConnectEvent) {}
func (o *clientObserver) OnSendQueue(wkclient.SendQueueEvent) {}
func (o *clientObserver) OnSendBatch(wkclient.SendBatchEvent) {}
func (o *clientObserver) OnSendAck(wkclient.SendAckEvent) {}
func (o *clientObserver) OnRecv(e wkclient.RecvEvent) {
	if e.Dropped {
		o.status.addRecv(0, 1)
		return
	}
	o.status.addRecv(1, 0)
}
func (o *clientObserver) OnError(e wkclient.ErrorEvent) {
	if e.Err != nil {
		o.status.addSendErrors(1, fmt.Sprintf("%s: %v", e.Op, e.Err))
	}
}

func sleepContext(ctx context.Context, d time.Duration) error {
	timer := time.NewTimer(d)
	defer timer.Stop()
	select {
	case <-ctx.Done():
		return ctx.Err()
	case <-timer.C:
		return nil
	}
}
```
